23 /**
24  * @brief Multiplies 2 matrices if possible otherwise it throws an error
25  *
26  * @tparam T Any numeric type
27  * @param A Matrix A to be multiplied in A * B
28  * @param B Matrix B to be multipled in A * B
29  * @return std::vector<std::vector<T>> Matrix C that is equal to A * B
30  * @throws Runtime Error if the 2 matrices cannot be multiplied
31  */
32 template <typename T, typename = typename std::enable_if<
33  std::is_arithmetic<T>::value, T>::type>
34 inline std::vector<std::vector<T>> MatrixMultiply(
35  const std::vector<std::vector<T>>& A,
36  const std::vector<std::vector<T>>& B) {
37  // If the 2 matrices cannot be multiplied throw a runtime error
38  if (!matrix_library::utils::matrix_utils::CanMatricesMultiply(A, B)) {
39  throw std::runtime_error("Matrices A and B cannot multiply");
40  }
41  // Create a matrix C with A number of rows and B number of columns
42  // Fill with default value of 0 to sum over
43  std::vector<std::vector<T>> C;
44  if (matrix_library::utils::matrix_utils::IsMatrixEmpty(B)) {
45  C.resize(A.size());
46  return C;
47  }
48  C = matrix_library::utils::matrix_utils::CreateMatrix(
49  A.size(), B.front().size(), static_cast<T>(0));
50 
51  // If C has no rows just return it
52  if (matrix_library::utils::matrix_utils::IsMatrixEmpty(C)) {
53  return C;
54  }
55  // If C has no colums just return it
56  if (matrix_library::utils::matrix_utils::IsVectorEmpty(C.front())) {
57  return C;
58  }
59  // Matrix multiplication done to be cache friendly
60  // In this the innermost loop variable is accessed in each row which is loaded
61  // contiguously in vector minimizing cache misses
62  // i loops over every row in A
63  // k loops over every column in A and row in B
64  // j loops over every column in B
65  for (size_t i = 0; i < A.size(); i++) {
66  for (size_t k = 0; k < A.front().size(); k++) {
67  for (size_t j = 0; j < B.front().size(); j++) {
68  C.at(i).at(j) += (A.at(i).at(k) * B.at(k).at(j));
69  }
70  }
71  }
72  return C;
73 }

75 /**
76  * @brief Transposes matrix if possible
77  *
78  * @tparam T Any numeric type
79  * @param original_matrix 2D matrix that we will make a transpose of
80  * @return std::vector<std::vector<T>> Transposed 2D matrix
81  * @throws Runtime error if the original matrix has no row
82  * @throws Runtime error if the original matrix has no columns
83  * @throws Runtime error if the original matrix has column count mismatch
84  */
85 template <typename T, typename = typename std::enable_if<
86  std::is_arithmetic<T>::value, T>::type>
87 inline std::vector<std::vector<T>> MatrixTranspose(
88  const std::vector<std::vector<T>>& original_matrix) {
89  // If matrix has no rows then this cannot be transposed
90  if (matrix_library::utils::matrix_utils::IsMatrixEmpty(original_matrix)) {
91  throw std::runtime_error("Original matrix has no rows");
92  }
93  // If matrix has no columns then this cannot be transposed
94  if (matrix_library::utils::matrix_utils::IsVectorEmpty(
95  original_matrix.front())) {
96  throw std::runtime_error("Original matrix has no column");
97  }
98  // If matrix has a column mismatch it cannot be transposed
99  if (!matrix_library::utils::matrix_utils::IsMatrixFollowingDimensions(
100  original_matrix, original_matrix.front().size())) {
101  throw std::runtime_error("Original matrix has a column mismatch");
102  }
103  // Initialize a transposed matrix with transposed dimensions
104  std::vector<std::vector<T>> transposed_matrix =
105  matrix_library::utils::matrix_utils::CreateMatrix(
106  original_matrix.front().size(), original_matrix.size(),
107  static_cast<T>(0));
108  // Iterating through matrix elements
109  for (size_t i = 0; i < original_matrix.size(); i++) {
110  for (size_t j = 0; j < original_matrix.front().size(); j++) {
111  // Storing element in flipped location in transposed matrix
112  transposed_matrix.at(j).at(i) = original_matrix.at(i).at(j);
113  }
114  }
115  // Return transposed matrix
116  return transposed_matrix;
117 }
