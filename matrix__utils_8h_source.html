23  *
24  * @tparam T Any numeric type
25  * @param num_rows Number of rows in matrix
26  * @param num_cols Number of columns in matrix
27  * @param value_to_fill Value to fill in matrix
28  * @return std::vector<std::vector<T>> 2D matrix with provided dimensions and

30  * @throws Runtime error if we are trying to make a matrix with 0 rows and non

32  * value

34 template <typename T, typename = typename std::enable_if<
35  std::is_arithmetic<T>::value, T>::type>
36 inline std::vector<std::vector<T>> CreateMatrix(size_t num_rows,
37  size_t num_cols,
38  T value_to_fill) {
39  // If number of rows is 0 and number of columns > 0 then this is impossible
40  if (num_rows == 0 && num_cols != 0) {
41  throw std::runtime_error("Impossible Matrix received");
42  }
43  // Initializing matrix
44  std::vector<std::vector<T>> matrix;
45  // Creating specified number of rows
46  matrix.resize(num_rows);
47  for (auto& row : matrix) {
48  // Resize fills vector with value as 2nd argument
49  row.resize(num_cols, value_to_fill);
50  }
51  return matrix;
52 }

54 /**
55  * @brief Create a Sequential Matrix of provided dimensions. Values filled
56  * based off start value. Each subsequent value in the row is then the previous
57  * value incremented by the increment value given. The values in new rows is
58  * based of the last value in the previous row + increment
59  *
60  * @tparam T Any numeric type
61  * @param num_rows Number of rows in matrix
62  * @param num_cols Number of columns in matrix
63  * @param start Base value to start with
64  * @param increment Value to increment with (Note: In unsigned matrices this
65  * can't be negative)
66  * @return std::vector<std::vector<T>> 2D matrix as specified
67  * @throws Runtime error if we are trying to make a matrix with 0 rows and non
68  * zero columns
69  */
70 template <typename T, typename = typename std::enable_if<
71  std::is_arithmetic<T>::value, T>::type>
72 inline std::vector<std::vector<T>> CreateSequentialMatrix(size_t num_rows,
73  size_t num_cols,
74  T start,
75  T increment) {
76  // If number of rows is 0 and number of columns > 0 then this is impossible
77  if (num_rows == 0 && num_cols != 0) {
78  throw std::runtime_error("Impossible Matrix received");
79  }
80  // Initializing matrix
81  std::vector<std::vector<T>> matrix;
82  matrix.resize(num_rows);
83  // Initializing value
84  T current_value = start;
85  for (auto& row : matrix) {
86  // Resizing to accommodate for values
87  row.resize(num_cols);
88  for (auto& elem : row) {
89  // Filling current value and then incrementing current value
90  elem = current_value;
91  current_value += increment;
92  }
93  }
94  return matrix;
95 }

146  * @brief Check if Matrix is following dimensions
147  *
148  * @tparam T Any numeric type
149  * @param matrix 2D matrix
150  * @param num_cols Number of columns supposed to be in matrix
151  * @return true If all rows have same number of columns as in parameter
152  * @return false If any one row does not have same number of columns as
153  * parameter
154  */
155 template <typename T, typename = typename std::enable_if<
156  std::is_arithmetic<T>::value, T>::type>
157 inline bool IsMatrixFollowingDimensions(
158  const std::vector<std::vector<T>>& matrix, size_t num_cols) {
159  // If matrix is empty and number of cols is 0 then it is following dimensions
160  if (IsMatrixEmpty(matrix) && num_cols == 0) {
161  return true;
162  // Else if matrix is empty but number of cols is not 0 then it isn't
163  // following dimensions
164  } else if (IsMatrixEmpty(matrix) && num_cols != 0) {
165  return false;
166  }
167  // For every row in matrix
168  for (const auto& row : matrix) {
169  // If it's size is not same as number of columns return false
170  if (row.size() != num_cols) {
171  std::cout << row.size() << " " << num_cols << std::endl;
172  return false;
173  }
174  }
175  // If we have reached here it means row sizes match columns and we have no
176  // mismatch
177  return true;
178 }

180 /**
181  * @brief Checks if 2 matrices can multiply
182  *
183  * @tparam T Any numeric type
184  * @param A Matrix A in A * B
185  * @param B Matrix B in A * B
186  * @return true If A and B can be multiplied
187  * @return false If A and B cannot be multiplied
188  */
189 template <typename T, typename = typename std::enable_if<
190  std::is_arithmetic<T>::value, T>::type>
191 inline bool CanMatricesMultiply(const std::vector<std::vector<T>>& A,
192  const std::vector<std::vector<T>>& B) {
193  // If Matrix A and B are empty they can be multiplied
194  if (IsMatrixEmpty(A) && IsMatrixEmpty(B)) {
195  return true;
196  }
197  // If Matrix A is empty it cannot be multiplied
198  if (IsMatrixEmpty(A)) {
199  std::cerr << "Matrix A is empty" << std::endl;
200  return false;
201  }
202  // If Matrix B is empty and A has 0 columns they can be multiplied
203  if (IsVectorEmpty(A.front()) && IsMatrixEmpty(B)) {
204  return true;
205  }
206  // If Matrix B is empty it cannot be multiplied
207  if (IsMatrixEmpty(B)) {
208  std::cerr << "Matrix B is empty" << std::endl;
209  return false;
210  }
211  // If Matrix A has no columns it cannot be multiplied
212  if (IsVectorEmpty(A.front())) {
213  std::cerr << "Matrix A has no columns" << std::endl;
214  return false;
215  }
216  // If Matrix A has a column mismatch it cannot be multiplied
217  if (!IsMatrixFollowingDimensions(A, A.front().size())) {
218  std::cerr << "Matrix A has a column mismatch" << std::endl;
219  return false;
220  }
221  // If Matrix B has a column mismatch it cannot be multiplied
222  if (!IsMatrixFollowingDimensions(B, B.front().size())) {
223  std::cerr << "Matrix B has a column mismatch" << std::endl;
224  return false;
225  }
226  // If Matrix A number of columns is not equal to Matrix B number of rows it
227  // cannot be multiplied
228  if (A.front().size() != B.size()) {
229  std::cerr << "Number of columns in Matrix A and number of rows in Matrix B "
230  "don't match"
231  << std::endl;
232  return false;
233  }
234  // If all conditions so far have been satisfied matrices can be multiplied
235  return true;
236 }

238 /**
239  * @brief Checks if 2 matrices have the same contents
240  *
241  * @tparam T Any numeric type
242  * @param A Matrix A
243  * @param B Matrix B
244  * @return true If both matrices have same contents
245  * @return false If both matrices do not have same contents
246  */
247 template <typename T, typename = typename std::enable_if<
248  std::is_arithmetic<T>::value, T>::type>
249 inline bool IsMatricesEqual(const std::vector<std::vector<T>>& A,
250  const std::vector<std::vector<T>>& B) {
251  // If the sizes are not equal they are not equal
252  if (A.size() != B.size()) {
253  return false;
254  }
255  // If the sizes are equal and they are both empty then they are equal
256  if (IsMatrixEmpty(A)) {
257  return true;
258  }
259  // If the column sizes are not equal they are not equal
260  if (A.front().size() != B.front().size()) {
261  return false;
262  }
263  // If both matrices are not following the given dimensions then they are
264  // ill-formed and cannot be determined if they are equal
265  if (!IsMatrixFollowingDimensions(A, A.front().size()) ||
266  !IsMatrixFollowingDimensions(B, B.front().size())) {
267  return false;
268  }
269  for (size_t i = 0; i < A.size(); i++) {
270  for (size_t j = 0; j < A.front().size(); j++) {
271  // If any of the elements are not equal then they are not equal
272  if (A.at(i).at(j) != B.at(i).at(j)) {
273  return false;
274  }
275  }
276  }
277  // If all conditions so far have passed then they are equal
278  return true;
279 }
